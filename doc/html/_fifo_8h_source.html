165  void resize() {
166  unsigned int newCap = (unsigned int) (capacity * SATURNIN_FIFO_UPDATE_FACTOR);
167  ASSERT((sizeof (T) * newCap) > (sizeof (T) * capacity));
168  T* tmp = (T*) malloc(newCap * sizeof (T));
169  ASSERT(tmp != nullptr);
170 #ifdef DEBUG
171  memset(tmp, 0, newCap * sizeof (T));
172 #endif /* DEBUG */
173  ASSERT_EQUAL(first, end);
174  memcpy(tmp, data + first, (capacity - first) * sizeof (T));
175  if (first > 0) {
176  memcpy(tmp + (capacity - first), data, end * sizeof (T));
177  }
178 #ifdef DEBUG
179  unsigned int k = first;
180  for (unsigned int i = 0; i < size; i++) {
181  ASSERT_EQUAL(data[k], tmp[i]);
182  k = (k + 1) % capacity;
183  }
184 #endif /* DEBUG */
185  free(data);
186  data = tmp;
187  first = 0;
188  end = size;
189  capacity = newCap;
190  }
