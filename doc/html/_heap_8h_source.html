235  void percolateUp(Var v) {
236  unsigned int pos = positions[v];
237  ASSERT(pos < data.getSize());
238  ASSERT_EQUAL(v, data[pos]);
239  bool stop = false;
240  while (!stop && pos > 0) {
241  unsigned int par = parent(pos);
242  if (comparator.compare(data[pos], data[par]) < 0) {
243  swap(data[par], data[pos]);
244  //update the position of the parent (which is now at pos)
245  positions[data[pos]] = pos;
246  pos = par;
247  } else {
248  stop = true;
249  }
250  }
251  positions[v] = pos;
252  data[pos] = v;
253  }
254 
260  void percolateDown(Var v) {
261  unsigned int pos = positions[v];
262  ASSERT(pos < data.getSize());
263  ASSERT_EQUAL(v, data[pos]);
264  unsigned int size = data.getSize();
265  bool stop = false;
266  while (!stop && left(pos) < size) {
267  unsigned int son = right(pos) < size &&
268  comparator.compare(data[right(pos)],
269  data[left(pos)]) < 0 ? right(pos) : left(pos);
270  if (comparator.compare(data[pos], data[son]) > 0) {
271  swap(data[pos], data[son]);
272  //update the position of the son (which is now at pos)
273  positions[data[pos]] = pos;
274  pos = son;
275  } else {
276  stop = true;
277  }
278  }
279  data[pos] = v;
280  positions[v] = pos;
281  }
