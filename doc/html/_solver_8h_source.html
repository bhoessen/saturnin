249  inline void printVar(Var v, bool colored = true, FILE* out = stdout) const {
250  colored = colored && assign.getSize() > v;
251  if (colored) {
252  if (assign[v] == wTrue) {
253  fprintf(out, "%s", VARIABLE_TRUE_COLOR);
254  } else if (assign[v] == wFalse) {
255  fprintf(out, "%s%s", VARIABLE_FALSE_COLOR, "-");
256  } else {
257  fprintf(out, "%s%s", VARIABLE_NO_COLOR, "?");
258  }
259  } else {
260  if (assign[v] == wFalse) {
261  fprintf(out, "-");
262  } else if (assign[v] == wUnknown) {
263  fprintf(out, "?");
264  }
265  }
266  fprintf(out, "%d", v);
267  if (colored) {
268  fprintf(out, "%s", VARIABLE_NO_COLOR);
269  }
270  }

280  inline void printLit(Lit l, bool colored = true, FILE* out = stdout) const {
281  Var v = VariablesManager::getVar(l);
282  colored = colored && assign.getSize() > v;
283  if (colored) {
284  if (VariablesManager::getLitSign(l) ?
285  assign[v] == wTrue : assign[v] == wFalse) {
286  fprintf(out, "%s", VARIABLE_TRUE_COLOR);
287  } else if (VariablesManager::getLitSign(l) ?
288  assign[v] == wFalse : assign[v] == wTrue) {
289  fprintf(out, "%s", VARIABLE_FALSE_COLOR);
290  } else {
291  fprintf(out, "%s", VARIABLE_NO_COLOR);
292  }
293  }
294  if (!VariablesManager::getLitSign(l)) {
295  fprintf(out, "-");
296  }
297  fprintf(out, "%d", v +1);
298  if (colored) {
299  fprintf(out, "%s", VARIABLE_NO_COLOR);
300  }
301  }
